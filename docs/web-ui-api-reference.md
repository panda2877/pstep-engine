## 四、SSE 消息类型定义

```typescript
// 计划阶段
interface PlanMessage {
  type: 'plan';
  content: string;
  steps: string[];
  totalSteps: number;
}

// 步骤执行
interface StepMessage {
  type: 'step';
  stepIndex: number;
  stepName: string;
  status: 'running' | 'completed' | 'failed';
  output?: string;
}

// 内容流式输出
interface ContentMessage {
  type: 'content';
  content: string;
  done?: boolean;
}

// 执行完成
interface DoneMessage {
  type: 'done';
  sessionId: string;
  messageCount: number;
  totalSteps: number;
  completedSteps: number;
  summary: string;
}

// 错误
interface ErrorMessage {
  type: 'error';
  message: string;
  code?: string;
}

// 联合类型
type SSEMessage = PlanMessage | StepMessage | ContentMessage | DoneMessage | ErrorMessage;
```

## 六、前端 API 封装示例

```typescript
// src/services/api.ts
import ky from 'ky';

const api = ky.create({
  prefixUrl: import.meta.env.VITE_API_BASE || '',
  timeout: 30000,
});

// Agent API
export const agentApi = {
  list: () => api.get('api/agents').json<{ agents: Agent[] }>(),
  get: (id: string) => api.get(`api/agents/${id}`).json<Agent>(),
  create: (data: CreateAgentRequest) => api.post('api/agents', { json: data }).json<Agent>(),
  update: (id: string, data: UpdateAgentRequest) => api.put(`api/agents/${id}`, { json: data }).json<Agent>(),
  delete: (id: string) => api.delete(`api/agents/${id}`).json(),
  sessions: (id: string) => api.get(`api/agents/${id}/sessions`).json<{ sessions: Session[] }>(),
};

// Session API
export const sessionApi = {
  list: (params: { projectId?: string; agentId?: string }) =>
    api.get('api/sessions', { searchParams: params }).json<{ sessions: Session[] }>(),
  get: (id: string) => api.get(`api/sessions/${id}`).json<Session & { messages: Message[] }>(),
  create: (data: CreateSessionRequest) => api.post('api/sessions', { json: data }).json<Session>(),
  delete: (id: string) => api.delete(`api/sessions/${id}`).json(),
};

// Chat API (SSE)
export const chatStream = (params: ChatRequest): EventSource => {
  const url = new URL('/api/chat', window.location.origin);
  Object.entries(params).forEach(([key, value]) => {
    if (value !== undefined) url.searchParams.set(key, String(value));
  });
  return new EventSource(url.toString());
};

// Memory API
export const memoryApi = {
  list: (params: { projectId: string; category?: string }) =>
    api.get('api/memory', { searchParams: params }).json<{ memories: MemoryEntry[] }>(),
  create: (data: CreateMemoryRequest) => api.post('api/memory', { json: data }).json<MemoryEntry>(),
  delete: (id: string) => api.delete(`api/memory/${id}`).json(),
};

// Message Search API
export const messageApi = {
  search: (params: { sessionId: string; query: string }) =>
    api.get('api/messages/search', { searchParams: params }).json<{ results: Message[]; total: number }>(),
};
```
